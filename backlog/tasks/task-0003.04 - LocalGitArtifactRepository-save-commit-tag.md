### Implementation Steps

#### Step 1: Implement `save()` method in LocalGitArtifactRepository

The save method must be wrapped in `writeQueue.submit(() -> { ... })` to serialise writes.

Inside the write queue task:

```java
public CommitResult save(Artifact artifact, String message, User user) {
    return writeQueue.submit(() -> doSave(artifact, message, user));
}

private CommitResult doSave(Artifact artifact, String message, User user) {
    ArtifactMetadata meta = artifact.metadata();
    String type = meta.type().folderName();   // e.g. "skills"
    String name = meta.name();                // e.g. "git-helper"
    String primaryFile = getPrimaryFileName(meta.type()); // e.g. "skill.md"
    String treePath = type + "/" + name + "/" + primaryFile;

    // 1. Encode artifact content (frontmatter + body)
    String fileContent = MetadataCodec.encode(artifact);
    byte[] contentBytes = fileContent.getBytes(StandardCharsets.UTF_8);

    // 2. Insert blob into object store
    ObjectId blobId = insertBlob(contentBytes);

    // 3. Build new tree (based on current HEAD tree, adding/replacing the file)
    ObjectId newTreeId = buildTree(treePath, blobId);

    // 4. Create commit
    String commitMsg = buildCommitMessage(meta, message);
    PersonIdent author = new PersonIdent(user.username(), user.username() + "@library");
    PersonIdent committer = new PersonIdent("agent-library", "server@agent-library");
    ObjectId commitId = createCommit(newTreeId, commitMsg, author, committer);

    // 5. Update HEAD ref
    updateHead(commitId);

    // 6. Create lightweight tag: <type>/<name>@<version>
    String tagName = meta.type().slug() + "/" + name + "@" + meta.version();
    createTag(tagName, commitId);

    // 7. Update INDEX.yaml (read current, add/update entry, commit again)
    updateIndex(meta, user);

    Instant timestamp = Instant.ofEpochSecond(repository.parseCommit(commitId).getCommitTime());
    return new CommitResult(commitId.name(), timestamp, commitMsg);
}
```

#### Step 3: JGit low-level API operations for bare repo

Since this is a bare repo (no working directory), we use low-level JGit APIs:

**Insert blob:**
```java
private ObjectId insertBlob(byte[] content) throws IOException {
    try (ObjectInserter inserter = repository.newObjectInserter()) {
        ObjectId blobId = inserter.insert(Constants.OBJ_BLOB, content);
        inserter.flush();
        return blobId;
    }
}
```

**Build tree (with nested directories):**
```java
private ObjectId buildTree(String path, ObjectId blobId) throws IOException {
    // Get current HEAD tree (or empty tree if no commits yet)
    ObjectId headTreeId = getHeadTree(); // nullable if empty repo
    
    try (ObjectInserter inserter = repository.newObjectInserter();
         TreeWalk tw = new TreeWalk(repository)) {
        // Use DirCacheBuilder approach:
        // 1. Create a DirCache (in-memory index)
        DirCache dc = DirCache.newInCore();
        DirCacheBuilder builder = dc.builder();
        
        // 2. If HEAD exists, copy all entries from HEAD tree
        if (headTreeId != null) {
            builder.addTree(new byte[0], DirCacheEntry.STAGE_0, 
                repository.newObjectReader(), headTreeId);
        }
        
        // 3. Add/replace our file entry
        DirCacheEntry entry = new DirCacheEntry(path);
        entry.setObjectId(blobId);
        entry.setFileMode(FileMode.REGULAR_FILE);
        builder.add(entry);
        
        builder.finish();
        
        // 4. Write the tree
        return dc.writeTree(inserter);
    }
}
```

**Create commit:**
```java
private ObjectId createCommit(ObjectId treeId, String message, 
        PersonIdent author, PersonIdent committer) throws IOException {
    try (ObjectInserter inserter = repository.newObjectInserter()) {
        CommitBuilder cb = new CommitBuilder();
        cb.setTreeId(treeId);
        cb.setMessage(message);
        cb.setAuthor(author);
        cb.setCommitter(committer);
        
        // Set parent to current HEAD (if exists)
        ObjectId headId = resolveHead();
        if (headId != null) {
            cb.setParentId(headId);
        }
        
        ObjectId commitId = inserter.insert(cb);
        inserter.flush();
        return commitId;
    }
}
```

**Update HEAD:**
```java
private void updateHead(ObjectId commitId) throws IOException {
    RefUpdate ru = repository.updateRef(Constants.HEAD);
    ru.setNewObjectId(commitId);
    ru.setRefLogMessage("commit", false);
    ObjectId oldHead = resolveHead();
    if (oldHead == null) {
        ru.setExpectedOldObjectId(ObjectId.zeroId());
    } else {
        ru.setExpectedOldObjectId(oldHead);
    }
    RefUpdate.Result result = ru.update();
    if (result != RefUpdate.Result.NEW && result != RefUpdate.Result.FAST_FORWARD) {
        throw new StorageException("Failed to update HEAD: " + result);
    }
}
```

**Create lightweight tag:**
```java
private void createTag(String tagName, ObjectId commitId) throws IOException {
    RefUpdate ru = repository.updateRef(Constants.R_TAGS + tagName);
    ru.setNewObjectId(commitId);
    ru.setRefLogMessage("tag", false);
    ru.setExpectedOldObjectId(ObjectId.zeroId());
    RefUpdate.Result result = ru.forceUpdate(); // forceUpdate allows overwrite for duplicate version
    if (result != RefUpdate.Result.NEW && result != RefUpdate.Result.FORCED 
            && result != RefUpdate.Result.NO_CHANGE) {
        throw new StorageException("Failed to create tag " + tagName + ": " + result);
    }
}
```

#### Step 4: Update INDEX.yaml

```java
private void updateIndex(ArtifactMetadata newMeta, User user) throws IOException {
    // 1. Read current INDEX.yaml content from HEAD tree
    String currentYaml = readFileFromHead("INDEX.yaml"); // returns "" if not found
    
    // 2. Parse to list
    List<ArtifactMetadata> entries = new ArrayList<>(IndexFile.load(currentYaml));
    
    // 3. Add or replace entry (match by type + name)
    entries.removeIf(e -> e.type() == newMeta.type() && e.name().equals(newMeta.name()));
    entries.add(newMeta);
    
    // 4. Serialize back
    String updatedYaml = IndexFile.save(entries);
    
    // 5. Commit INDEX.yaml update
    byte[] indexBytes = updatedYaml.getBytes(StandardCharsets.UTF_8);
    ObjectId blobId = insertBlob(indexBytes);
    ObjectId treeId = buildTree("INDEX.yaml", blobId);
    PersonIdent serverIdent = new PersonIdent("agent-library", "server@agent-library");
    String indexMsg = "chore: update INDEX.yaml";
    ObjectId commitId = createCommit(treeId, indexMsg, serverIdent, serverIdent);
    updateHead(commitId);
}
```

#### Step 6: Helper to read file from HEAD tree

```java
private String readFileFromHead(String path) throws IOException {
    ObjectId headId = resolveHead();
    if (headId == null) return "";
    
    try (RevWalk rw = new RevWalk(repository)) {
        RevCommit commit = rw.parseCommit(headId);
        RevTree tree = commit.getTree();
        
        try (TreeWalk tw = TreeWalk.forPath(repository, path, tree)) {
            if (tw == null) return ""; // file not found
            ObjectId blobId = tw.getObjectId(0);
            byte[] bytes = repository.open(blobId).getBytes();
            return new String(bytes, StandardCharsets.UTF_8);
        }
    }
}
```

#### Step 7: resolveHead() helper

```java
private ObjectId resolveHead() throws IOException {
    Ref headRef = repository.exactRef(Constants.HEAD);
    if (headRef == null) return null;
    return headRef.getObjectId(); // null if HEAD exists but points to unborn branch
}

private ObjectId getHeadTree() throws IOException {
    ObjectId headId = resolveHead();
    if (headId == null) return null;
    try (RevWalk rw = new RevWalk(repository)) {
        return rw.parseCommit(headId).getTree().getId();
    }
}
```
